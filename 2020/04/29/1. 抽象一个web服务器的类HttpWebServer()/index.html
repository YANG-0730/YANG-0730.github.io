# 返回指定页面完整多任务面向对象版本

# 1. 实现tcp服务器
# 1.2 等待客户端链接，一旦链接创建一个子线程处理客户端的请求:
    # 2. 接收浏览器发送过来的HTTP请求报文, 获取请求资源路径
    # 3. 读取指定页面(static/请求资源路径), 拼接成HTTP响应报文返回给浏览器
    # 4. 关闭服务浏览器的socket


# 1.抽象一个web服务器的类 HttpWebServer()
# 2.类的初始化方法，需要对属性socket初始化
# 3.类的启动方法， 实现服务器的启动，并提供服务


# 1、实现tcp服务器
import socket
import threading


# 1. 抽象一个web服务器的类HttpWebServer()
class HttpWebServer(object):
    # 2.类的初始化方法，需要对属性socket初始化
    def __init__(self):
        # 1. 创建服务器socket
        self.tcp_server_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        # 2. 设置端口复用
        self.tcp_server_socket.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, True)
        # 3. 绑定端口
        self.tcp_server_socket.bind(('', 8080))
        # 4. 设置监听模式
        self.tcp_server_socket.listen(128)

    # 3.类的启动方法， 实现服务器的启动，并提供服务
    def start(self):
        while True:
            # 5.1 等待客户端链接
            new_socket, client_ip_port = self.tcp_server_socket.accept()
            # 创建子线程对象
            sub_thread = threading.Thread(target=self.handle_request, args=(new_socket, client_ip_port))
            # 设置子线程守护主线程,防止主线程无法结束
            sub_thread.daemon = True
            # sub_thread.setDaemon(True)
            # 启动子线程
            sub_thread.start()

    @staticmethod
    # 处理浏览器请求函数
    def handle_request(new_socket, client_ip_port):
        # 5.2 接收浏览器发送过来的HTTP请求
        recv_data = new_socket.recv(4096)
        # print('请求报文二进制格式:')
        # print(recv_data)
        #
        # print('请求报文字符串格式:')
        # print(recv_data.decode('utf-8'))

        # 注意:浏览器不会长期维护链接,会主断掉链接,服务器的recv解阻塞,返回长度为0的数据
        if len(recv_data) == 0:
            print(client_ip_port, '浏览器关闭链接了...')
            new_socket.close()
            return

        recv_str = recv_data.decode('utf-8')
        recv_list = recv_str.split(' ', maxsplit=2)
        request_path = recv_list[1]
        print('请求资源路径:', request_path)

        # 设置默认首页
        if request_path == '/':
            request_path = '/index.html'

        try:
            # 3. 读取指定页面(static/请求资源路径), 拼接成HTTP响应报文返回给浏览器
            with open('static' + request_path, 'rb') as file:
                file_data = file.read()
        except Exception as e:
            '''请求资源路径不存在'''
            # 返回一个404页面
            response_line = 'HTTP/1.1 404 Not Found\r\n'
            response_head = 'Server: YANG/1.0\r\n'
            with open('static/error.html', 'rb') as file:
                file_data = file.read()
            response_body = file_data
            response_data = (response_line + response_head + '\r\n').encode('utf-8') + response_body
            new_socket.send(response_data)
        else:
            '''请求资源路径存在'''
            # 3.1 拼接成HTTP响应报文
            # 响应行
            response_line = 'HTTP/1.1 200 OK\r\n'
            # 响应头
            response_head = 'Server: YANG/1.0\r\n'
            # 空行
            # \r\n
            # 响应体
            response_body = file_data
            # 拼接成HTTP响应报文
            response_data = (response_line + response_head + '\r\n').encode('utf-8') + response_body
            # 返回给浏览器
            new_socket.send(response_data)
        finally:
            # 4.关闭服务浏览器的socket
            new_socket.close()


def main():
    # 使用我们web服务器类实例化对象
    wb = HttpWebServer()
    # 启动服务器
    wb.start()


if __name__ == '__main__':
    main()
